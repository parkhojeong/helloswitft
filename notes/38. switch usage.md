* switch pattern matching
- <, > is not supported 
```swift
var age = 20
switch age {
    case 10...19: print("10~19")
    case 20...29: print("20~29")
    default: print("default")
}
```

* switch and value binding
```swift
let month = 10
switch month {
    case let a: print(a)
    default: break
}
```

* switch and where clause
```swift
var num = 7
switch num {
    case let x where x % 2 == 0: print("even \(x)")
    case let x where x % 2 != 0: print("odd \(x)")
    default: break
}
```
