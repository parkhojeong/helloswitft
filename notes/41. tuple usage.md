* tuple matching
```swift
let ios = (language: "Swift", version: "5")
if ios == ("Swift", "5") {
    // code
}

switch ios {
    case ("Swift", "5"): print("print")
    default: break
}
```

- tuple binding
```swift
let coordinate = (0, 5)
switch coordinate {
    case (let x, 0), (0, let x): print("x or y is 0")
    default: break
}
```

- tuple usage
```swift
let coordinate = (1, -1)
switch coordinate {
    case let (x,y) where x == y: print("x == y")
    case let (x,y) where x == -y: print("x == -y")
    default: break
}
```
